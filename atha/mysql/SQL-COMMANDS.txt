### CREATE A TABLE TRANSACTIONS WITH FOREIGN_KEY RELATIONSHIP TO PRODUCT AND CUSTOMER

CREATE TABLE transactions (id INT(3) PRIMARY KEY AUTO_INCREMENT NOT NULL, 
                       			customer_id INT(5),
                       			product_id INT(5),
                       			qty INT(5), 
                       			_date DATE(),
                          		FOREIGN KEY (customer_id) REFERENCES customers(id),
                           		FOREIGN KEY (product_id) REFERENCES products(id)
                          )

### GET TOTAL NUMBER OF PURCHASES, AND NUMBER OF TIMES PURCHASE WAS MADE FOR EACH ITEM IN TABLE AND JOIN THE CORRESPONDING PRODUCT NAME FROM THE PRODUCTS TABLE (W3SHOOLS DATABASE.)

SELECT ProductName, SUM(Quantity) AS TOTAL, COUNT(ProductName) AS PURCHASES FROM (SELECT * FROM OrderDetails
JOIN Products AS T
On T.ProductID = OrderDetails.ProductID)
GROUP BY ProductName
ORDER BY SUM(Quantity) DESC;
